    PROCEDURE CalcInventoryValue@1150001();
    VAR
      CostExpPerEntry@1150000 : Decimal;
      CostActPerEntry@1150001 : Decimal;
      CostToGLPerEntry@1150004 : Decimal;
      InvQuantityPerEntry@1150002 : Decimal;
      TmpInvPostingGr@1150003 : Code[10];
    BEGIN
      WITH ValueEntry DO BEGIN
        SETCURRENTKEY("Item Ledger Entry No.");
        SETRANGE("Item Ledger Entry No.","Item Ledger Entry"."Entry No.");
        SETRANGE("Posting Date",0D,StatusDate);

        IF FIND('-') THEN BEGIN
          REPEAT
            TmpInvPostingGr := "Inventory Posting Group";
            CostActPerEntry := CostActPerEntry + "Cost Amount (Actual)";
            CostExpPerEntry := CostExpPerEntry + "Cost Amount (Expected)";
            InvQuantityPerEntry := InvQuantityPerEntry + "Invoiced Quantity";
            CostToGLPerEntry := CostToGLPerEntry + "Cost Posted to G/L";
          UNTIL NEXT = 0;

          IF NOT PostingGroupBuffer.GET(TmpInvPostingGr) THEN BEGIN
            PostingGroupBuffer.INIT;
            PostingGroupBuffer.Code := TmpInvPostingGr;
          END;

          PostingGroupBuffer."Inv. Valuation Actual" := PostingGroupBuffer."Inv. Valuation Actual" + CostActPerEntry;
          PostingGroupBuffer."Inv. Posted to GL" := PostingGroupBuffer."Inv. Posted to GL" + CostToGLPerEntry;
          CostActual := CostActual + CostActPerEntry;
          CostPostedToGL := CostPostedToGL + CostToGLPerEntry;
          InvQuantity := InvQuantity + InvQuantityPerEntry;

          IF InvQuantityPerEntry = "Item Ledger Entry".Quantity THEN BEGIN
            CostExpected := CostExpected + CostActPerEntry;
            PostingGroupBuffer."Inv. Valuation Exp." := PostingGroupBuffer."Inv. Valuation Exp." + CostActPerEntry;
          END ELSE BEGIN
            CostExpected := CostExpected + CostExpPerEntry + CostActPerEntry;
            PostingGroupBuffer."Inv. Valuation Exp." := PostingGroupBuffer."Inv. Valuation Exp." + CostExpPerEntry + CostActPerEntry;
          END;

          IF NOT PostingGroupBuffer.MODIFY THEN
            PostingGroupBuffer.INSERT;
        END;
      END;
    END;

    BEGIN
    END.
