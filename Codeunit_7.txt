              GLBudgetName.SETRANGE(Name);

              IF NOT GLBudgetName.FINDFIRST THEN BEGIN
                GLBudgetName.INIT;
                GLBudgetName.Name := Text000;
                GLBudgetName.Description := Text001;
                GLBudgetName.INSERT;
              END;
              SETFILTER("Budget Filter",GLBudgetName.Name);
            END;
          END;
